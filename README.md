# @codes-on-us/assistant

## Installation

```
$ npm i @codes-on-us/assistant
```

## Features

- User Assistant
- HTTP Assistant


## Usage

```jsx
import { AssistantProvicer } from '@codes-on-us/assistant';
 
const root = ReactDOM.createRoot(
	document.getElementById('root') as HTMLElement
 );

root.render(
  <AssistantProvicer>
    <App />
  </AssistantProvicer>
);
```
